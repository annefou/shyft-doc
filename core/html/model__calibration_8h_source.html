427  void prepare_optimize() {
428  // 1. ensure vectors reflects current state of lower..upper bounds
429  p_min = p_vector(parameter_lower_bound);
430  p_max = p_vector(parameter_upper_bound);
431  // 2. figure out n_catchments, asking the model
432  n_catchments = model.number_of_catchments();
433  // 3. figure out the catchment indexes to evaluate..
434  // and if we need to turn on snow collection
435  vector<int> catchment_indexes;
436  model.set_snow_sca_swe_collection(-1, false);//turn off all snow by default.
437  for (const auto&t : targets) {
438  catchment_indexes.insert(catchment_indexes.end(), begin(t.catchment_indexes), end(t.catchment_indexes));
439  if (t.catchment_property != DISCHARGE)
440  for (auto cid : t.catchment_indexes)
441  model.set_snow_sca_swe_collection(cid, true);//turn on for those with something like snow enabled
442  }

444  sort(begin(catchment_indexes), end(catchment_indexes));

446  catchment_indexes.resize(distance(begin(catchment_indexes), unique_end));
447  }

583  vector<double> to_scaled(const vector<double>& rp) const {
584  if (p_min.size() == 0) throw runtime_error("Parameter ranges are not set");

589  p_s.reserve(n_params);

698  double run(const vector<double>& rp) {
699  auto p = expand_p_vector(rp);// expand to full vector, then:
700  parameter_accessor.set(p); // Sets global parameters, all cells share a common pointer.
701  reset_states();
702  model.run_cells();
703  double goal_function_value = 0.0;// overall goal-function, intially zero
704  double scale_factor_sum = 0.0; // each target-spec have a weight, -use this to sum up weights
705  vector<pts_t> catchment_d;
706  vector<area_ts> catchment_sca, catchment_swe;// "catchment" level simulated discharge,sca,swe
707  for (const auto& t : targets) {
708  shyft::timeseries::direct_accessor<pts_t, timeaxis_t> target_accessor(t.ts, t.ts.ta);
709  pts_t property_sum;
710  switch (t.catchment_property) {
711  case DISCHARGE:
712  property_sum = compute_discharge_sum(t, catchment_d);
713  break;
714  case SNOW_COVERED_AREA:
715  property_sum = compute_sca_sum(t, catchment_sca);
716  break;
717  case SNOW_WATER_EQUIVALENT:
718  property_sum = compute_swe_sum(t, catchment_swe);
719  break;
720  }
721  shyft::timeseries::average_accessor<pts_t, timeaxis_t> property_sum_accessor(property_sum, t.ts.ta);
722  double partial_goal_function_value;
723  if (t.calc_mode == target_spec_calc_type::NASH_SUTCLIFFE) {
724  partial_goal_function_value = nash_sutcliffe_goal_function(target_accessor, property_sum_accessor);
725  } else {
726  // ref. KLING-GUPTA Journal of Hydrology 377 (2009) 8091, page 83, formula (10):
727  // a=alpha, b=betha, q =sigma, u=my, s=simulated, o=observed
728  partial_goal_function_value /* EDs */ = kling_gupta_goal_function<dlib::running_scalar_covariance<double>>(target_accessor,
729  property_sum_accessor,
730  t.s_r,
731  t.s_a,
732  t.s_b);
733  }
734  if (isfinite(partial_goal_function_value)) {
735  scale_factor_sum += t.scale_factor;
736  goal_function_value += t.scale_factor*partial_goal_function_value;
737  } else if (print_progress_level > 0) {
738  cout << "warning: goal-function " << t.catchment_property << ": evaluated as nan" << endl;
739  }
740  }
741  goal_function_value /= scale_factor_sum;
742  if (print_progress_level > 0) {
743  cout << "ParameterVector(";
744  for (size_t i = 0; i < parameter_accessor.size(); ++i) {
745  cout << parameter_accessor.get(i);
746  if (i < parameter_accessor.size() - 1)cout << ", ";
747  }
748  cout << ") = " << goal_function_value << " (NS or KG)" << endl;
749  }
750  return goal_function_value;
751  }
