690  double run(const vector<double>& rp) {
691  auto p = expand_p_vector(rp);// expand to full vector, then:
692  parameter_accessor.set(p); // Sets global parameters, all cells share a common pointer.
693  reset_states();
694  model.run_cells();
695  double goal_function_value = 0.0;// overall goal-function, intially zero
696  double scale_factor_sum = 0.0; // each target-spec have a weight, -use this to sum up weights
697  vector<pts_t> catchment_d;
698  vector<area_ts> catchment_sca, catchment_swe;// "catchment" level simulated discharge,sca,swe
699  for (const auto& t : targets) {
700  shyft::timeseries::direct_accessor<pts_t, timeaxis_t> target_accessor(t.ts, t.ts.ta);
701  pts_t property_sum;
702  switch (t.catchment_property) {
703  case DISCHARGE:
704  property_sum = compute_discharge_sum(t, catchment_d);
705  break;
706  case SNOW_COVERED_AREA:
707  property_sum = compute_sca_sum(t, catchment_sca);
708  break;
709  case SNOW_WATER_EQUIVALENT:
710  property_sum = compute_swe_sum(t, catchment_swe);
711  break;
712  }
713  shyft::timeseries::average_accessor<pts_t, timeaxis_t> property_sum_accessor(property_sum, t.ts.ta);
714  double partial_goal_function_value;
715  if (t.calc_mode == target_spec_calc_type::NASH_SUTCLIFFE) {
716  partial_goal_function_value = nash_sutcliffe_goal_function(target_accessor, property_sum_accessor);
717  } else {
718  // ref. KLING-GUPTA Journal of Hydrology 377 (2009) 8091, page 83, formula (10):
719  // a=alpha, b=betha, q =sigma, u=my, s=simulated, o=observed
720  partial_goal_function_value /* EDs */ = kling_gupta_goal_function<dlib::running_scalar_covariance<double>>(target_accessor,
721  property_sum_accessor,
722  t.s_r,
723  t.s_a,
724  t.s_b);
725  }
726  if (isfinite(partial_goal_function_value)) {
727  scale_factor_sum += t.scale_factor;
728  goal_function_value += t.scale_factor*partial_goal_function_value;
729  } else if (print_progress_level > 0) {
730  cout << "warning: goal-function " << t.catchment_property << ": evaluated as nan" << endl;
731  }
732  }
733  goal_function_value /= scale_factor_sum;
734  if (print_progress_level > 0) {
735  cout << "ParameterVector(";
736  for (size_t i = 0; i < parameter_accessor.size(); ++i) {
737  cout << parameter_accessor.get(i);
738  if (i < parameter_accessor.size() - 1)cout << ", ";
739  }
740  cout << ") = " << goal_function_value << " (NS or KG)" << endl;
741  }
742  return goal_function_value;
743  }
