238  void run_hbv_stack(const GCD& geo_cell_data,
239  const P& parameter,
240  const T& time_axis, int start_step, int n_steps,
241  const T_TS& temp,
242  const P_TS& prec,
243  const WS_TS& wind_speed,
244  const RH_TS& rel_hum,
245  const RAD_TS& rad,
246  S& state,
247  SC& state_collector,
248  RC& response_collector
249  ) {
250  // Access time series input data through accessors of template A (typically a direct accessor).
251  using temp_accessor_t = A<T_TS, T>;
252  using prec_accessor_t = A<P_TS, T>;
253  //using wind_speed_accessor_t = A<WS_TS, T>;
254  using rel_hum_accessor_t = A<RH_TS, T>;
255  using rad_accessor_t = A<RAD_TS, T>;
256 
257  auto temp_accessor = temp_accessor_t(temp, time_axis);
258  auto prec_accessor = prec_accessor_t(prec, time_axis);
259  // not used:auto wind_speed_accessor = wind_speed_accessor_t(wind_speed, time_axis);
260  auto rel_hum_accessor = rel_hum_accessor_t(rel_hum, time_axis);
261  auto rad_accessor = rad_accessor_t(rad, time_axis);
262 
263  // Initialize the method stack
264  precipitation_correction::calculator p_corr(parameter.p_corr.scale_factor);
265  priestley_taylor::calculator pt(parameter.pt.albedo, parameter.pt.alpha);
266  hbv_snow::calculator<typename P::snow_parameter_t, typename S::snow_state_t> snow(parameter.snow, state.snow);
267  hbv_soil::calculator<typename P::soil_parameter_t> soil(parameter.soil);
268  hbv_tank::calculator<typename P::tank_parameter_t> tank(parameter.tank);
269  R response;
270  const double total_lake_fraction = geo_cell_data.land_type_fractions_info().lake() + geo_cell_data.land_type_fractions_info().reservoir();
271  const double glacier_fraction = geo_cell_data.land_type_fractions_info().glacier();
272  const double land_fraction = 1 - total_lake_fraction - glacier_fraction;
273  const double glacier_area_m2 = geo_cell_data.area()*glacier_fraction;//const double forest_fraction = geo_cell_data.land_type_fractions_info().forest();
274 
275  size_t i_begin = n_steps > 0 ? start_step : 0;
276  size_t i_end = n_steps > 0 ? start_step + n_steps : time_axis.size();
277  for (size_t i = i_begin; i < i_end; ++i) {
278  utcperiod period = time_axis.period(i);
279  double temp = temp_accessor.value(i);
280  double rad = rad_accessor.value(i);
281  double rel_hum = rel_hum_accessor.value(i);
282  double prec = p_corr.calc(prec_accessor.value(i));
283  state_collector.collect(i, state);
284 
285  snow.step(state.snow, response.snow, period.start, period.end, parameter.snow, prec, temp);
286 
287  response.gm_melt_m3s = glacier_melt::step(parameter.gm.dtf,temp,geo_cell_data.area()*state.snow.sca,glacier_area_m2);// m3/s, that is, how much flow from the snow free glacier parts
288  response.pt.pot_evapotranspiration = pt.potential_evapotranspiration(temp, rad, rel_hum)*calendar::HOUR; // mm/h
289  response.ae.ae = hbv_actual_evapotranspiration::calculate_step(
290  state.soil.sm, response.pt.pot_evapotranspiration,
291  parameter.ae.lp, std::max(state.snow.sca,glacier_fraction), // a evap only on non-snow/non-glac area
292  period.timespan());
293 
294  soil.step(state.soil, response.soil, period.start, period.end, response.snow.outflow, response.ae.ae);
295 
296  tank.step(state.tank, response.tank, period.start, period.end, response.soil.outflow);
297 
298  response.total_discharge =
299  prec*total_lake_fraction // precipitation on lake|reservoirs give direct response in this alg.
300  + m3s_to_mmh(response.gm_melt_m3s, geo_cell_data.area()) // the glacier also direct
301  + response.tank.outflow * land_fraction;
302 
303  // Possibly save the calculated values using the collector callbacks.
304  response_collector.collect(i, response);
305  if (i + 1 == i_end)
306  state_collector.collect(i + 1, state);
307  }
308  response_collector.set_end_response(response);
309  }
